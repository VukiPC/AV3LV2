c) public class JednakokracanTrokut : Trokut {
    public double A { get; set; } // krak
    public double B { get; set; } // osnovica

    private double Visina() {
        return Math.Sqrt(Math.Pow(A, 2) - Math.Pow(B / 2, 2));
    }

    public bool isValid() {
        return (2 * A > B); // Trokuev nejednakost
    }

    public double Povrsina() => (B * Visina()) / 2;
}
